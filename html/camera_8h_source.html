179 void initialize() {
180 imgheight = static_cast<int>(imgwidth / aspectRatio);
181 imgheight = (imgheight < 1) ? 1 : imgheight;
182
183 cameraCentre = cameraPosition;
184
185 // viewport
186 auto focalLength = (cameraPosition - cameraLookAt).length();
187 auto theta = fov * pi / 180.0;
188 auto h = tan(theta/2);
189
190 auto viewportHEIGHT = 2.0 * h * focalLength;
191 auto viewportWidth = viewportHEIGHT * (static_cast<double>(imgwidth)/imgheight);
192
193 w = normalize(cameraPosition - cameraLookAt);
194 u = normalize(cross(cameraUp, w));
195 v = cross(w, u);
196
197 vec3 viewportU = viewportWidth * u;
198 vec3 viewportV = viewportHEIGHT * -v;
199
200 pixelDeltaU = viewportU/ imgwidth;
201 pixelDeltaV = viewportV / imgheight;
202
203 auto viewportUpperLeft = cameraCentre - (focalLength * w) - viewportU/2 - viewportV/2;
204 pixelUpperLeft = viewportUpperLeft + 0.5 * (pixelDeltaU + pixelDeltaV);
205 }

207 ray getRay(int i, int j) const {
208
209 auto pixelCentre= pixelUpperLeft + (i * pixelDeltaU) + (j * pixelDeltaV);
210 auto pixelSample= pixelCentre + randomPSample();
211
212 auto rayOrigin = cameraCentre;
213 auto rayDirection = pixelSample - cameraCentre; // direction from ray origin to randomly generated pixel sample
214 return ray(rayOrigin, rayDirection); // return ray that points to random sample
215 }
216
217 vec3 randomPSample() const {
218 // generate a random sample in the pixel square
219 auto px = -0.5 + RandomDouble();
220 auto py = -0.5 + RandomDouble();
221 return (px * pixelDeltaU) + (py * pixelDeltaV);
222 }

224 colour getRayColour(const ray& r, int bounces, const hittable& scene) {
225 hitRecord rec;
226
227 if (bounces<= 0)
228 return colour(0,0,0);
229
230
231 if (!scene.hit(r, interval(0.00001, infinity), rec)) {
232 vec3 nDir = normalize(r.direction()); // return sky if no hit
233 auto offset = 0.5*(nDir.y() + 1.0);
234 return (1.0-offset)*colour(1.0, 1.0, 1.0) + offset*colour(0.357, 0.608, 0.988);
235 }
236
237
238 ray scattered;
239 colour attenuation;
240 colour lightColour = rec.m->emitted(rec.u, rec.v, rec.p);
241
242 if (!rec.m->scatter(r, rec, attenuation, scattered))
243 return lightColour;
244
245 colour accumulatedColour = attenuation * getRayColour(scattered, bounces-1, scene);
246
247 return lightColour + accumulatedColour;
